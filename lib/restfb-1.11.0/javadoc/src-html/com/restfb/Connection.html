023package com.restfb;
024
025import static com.restfb.util.StringUtils.isBlank;
026import static java.util.Collections.unmodifiableList;
027
028import java.util.ArrayList;
029import java.util.Iterator;
030import java.util.List;
031import java.util.NoSuchElementException;
032
033import com.restfb.exception.FacebookJsonMappingException;
034import com.restfb.json.JsonArray;
035import com.restfb.json.JsonException;
036import com.restfb.json.JsonObject;
037import com.restfb.util.ReflectionUtils;
038
039/**
040 * Represents a <a href="http://developers.facebook.com/docs/api">Graph API Connection type</a>.
041 * 
042 * @author <a href="http://restfb.com">Mark Allen</a>
043 */
044public class Connection<T> implements Iterable<List<T>> {
045  private FacebookClient facebookClient;
046  private Class<T> connectionType;
047  private List<T> data;
048  private String previousPageUrl;
049  private String nextPageUrl;
050  private Long totalCount;
051  private String beforeCursor;
052  private String afterCursor;
053
054  /**
055   * @see java.lang.Iterable#iterator()
056   * @since 1.6.7
057   */
058  @Override
059  public Iterator<List<T>> iterator() {
060    return new ConnectionIterator<T>(this);
061  }
062
063  /**
064   * Iterator over connection pages.
065   * 
066   * @author <a href="http://restfb.com">Mark Allen</a>
067   * @since 1.6.7
068   */
069  protected static class ConnectionIterator<T> implements Iterator<List<T>> {
070    private Connection<T> connection;
071    private boolean initialPage = true;
072
073    /**
074     * Creates a new iterator over the given {@code connection}.
075     * 
076     * @param connection
077     *          The connection over which to iterate.
078     */
079    protected ConnectionIterator(Connection<T> connection) {
080      this.connection = connection;
081    }
082
083    /**
084     * @see java.util.Iterator#hasNext()
085     */
086    @Override
087    public boolean hasNext() {
088      // Special case: initial page will always have data
089      if (initialPage) {
090        return true;
091      }
092
093      return connection.hasNext();
094    }
095
096    /**
097     * @see java.util.Iterator#next()
098     */
099    @Override
100    public List<T> next() {
101      // Special case: initial page will always have data, return it
102      // immediately.
103      if (initialPage) {
104        initialPage = false;
105        return connection.getData();
106      }
107
108      if (!connection.hasNext()) {
109        throw new NoSuchElementException("There are no more pages in the connection.");
110      }
111        
112      connection = connection.fetchNextPage();
113      return connection.getData();
114    }
115
116    /**
117     * @see java.util.Iterator#remove()
118     */
119    @Override
120    public void remove() {
121      throw new UnsupportedOperationException(ConnectionIterator.class.getSimpleName()
122          + " doesn't support the remove() operation.");
123    }
124  }
125
126  /**
127   * Creates a connection with the given {@code jsonObject}.
128   * 
129   * @param facebookClient
130   *          The {@code FacebookClient} used to fetch additional pages and map data to JSON objects.
131   * @param json
132   *          Raw JSON which must include a {@code data} field that holds a JSON array and optionally a {@code paging}
133   *          field that holds a JSON object with next/previous page URLs.
134   * @param connectionType
135   *          Connection type token.
136   * @throws FacebookJsonMappingException
137   *           If the provided {@code json} is invalid.
138   * @since 1.6.7
139   */
140  @SuppressWarnings("unchecked")
141  public Connection(FacebookClient facebookClient, String json, Class<T> connectionType) {
142    List<T> data = new ArrayList<T>();
143
144    if (json == null) {
145      throw new FacebookJsonMappingException("You must supply non-null connection JSON.");
146    }
147
148    JsonObject jsonObject = null;
149
150    try {
151      jsonObject = new JsonObject(json);
152    } catch (JsonException e) {
153      throw new FacebookJsonMappingException("The connection JSON you provided was invalid: " + json, e);
154    }
155
156    // Pull out data
157    JsonArray jsonData = jsonObject.getJsonArray("data");
158    for (int i = 0; i < jsonData.length(); i++) {
159      data.add(connectionType.equals(JsonObject.class) ? (T) jsonData.get(i) : facebookClient.getJsonMapper()
160        .toJavaObject(jsonData.get(i).toString(), connectionType));
161    }
162
163    // Pull out paging info, if present
164    if (jsonObject.has("paging")) {
165      JsonObject jsonPaging = jsonObject.getJsonObject("paging");
166      previousPageUrl = jsonPaging.has("previous") ? jsonPaging.getString("previous") : null;
167      nextPageUrl = jsonPaging.has("next") ? jsonPaging.getString("next") : null;
168      if (null != previousPageUrl && previousPageUrl.startsWith("http://")) {
169        previousPageUrl = previousPageUrl.replaceFirst("http://", "https://");
170      }
171      if (null != nextPageUrl && nextPageUrl.startsWith("http://")) {
172        nextPageUrl = nextPageUrl.replaceFirst("http://", "https://");
173      }
174    } else {
175      previousPageUrl = null;
176      nextPageUrl = null;
177    }
178
179    if (jsonObject.has("paging") && jsonObject.getJsonObject("paging").has("cursors")) {
180      JsonObject jsonCursors = jsonObject.getJsonObject("paging").getJsonObject("cursors");
181      beforeCursor = jsonCursors.has("before") ? jsonCursors.getString("before") : null;
182      afterCursor = jsonCursors.has("after") ? jsonCursors.getString("after") : null;
183    }
184
185    if (jsonObject.has("summary")) {
186      JsonObject jsonSummary = jsonObject.getJsonObject("summary");
187      totalCount = jsonSummary.has("total_count") ? jsonSummary.getLong("total_count") : null;
188    } else {
189      totalCount = null;
190    }
191
192    this.data = unmodifiableList(data);
193    this.facebookClient = facebookClient;
194    this.connectionType = connectionType;
195  }
196
197  /**
198   * Fetches the next page of the connection. Designed to be used by {@link ConnectionIterator}.
199   * 
200   * @return The next page of the connection.
201   * @since 1.6.7
202   */
203  protected Connection<T> fetchNextPage() {
204    return facebookClient.fetchConnectionPage(getNextPageUrl(), connectionType);
205  }
206
207  @Override
208  public String toString() {
209    return ReflectionUtils.toString(this);
210  }
211
212  @Override
213  public boolean equals(Object object) {
214    return ReflectionUtils.equals(this, object);
215  }
216
217  @Override
218  public int hashCode() {
219    return ReflectionUtils.hashCode(this);
220  }
221
222  /**
223   * Data for this connection.
224   * 
225   * @return Data for this connection.
226   */
227  public List<T> getData() {
228    return data;
229  }
230
231  /**
232   * This connection's "previous page of data" URL.
233   * 
234   * @return This connection's "previous page of data" URL, or {@code null} if there is no previous page.
235   * @since 1.5.3
236   */
237  public String getPreviousPageUrl() {
238    return previousPageUrl;
239  }
240
241  /**
242   * This connection's "next page of data" URL.
243   * 
244   * @return This connection's "next page of data" URL, or {@code null} if there is no next page.
245   * @since 1.5.3
246   */
247  public String getNextPageUrl() {
248    return nextPageUrl;
249  }
250
251  /**
252   * Does this connection have a previous page of data?
253   * 
254   * @return {@code true} if there is a previous page of data for this connection, {@code false} otherwise.
255   */
256  public boolean hasPrevious() {
257    return !isBlank(getPreviousPageUrl());
258  }
259
260  /**
261   * Does this connection have a next page of data?
262   * 
263   * @return {@code true} if there is a next page of data for this connection, {@code false} otherwise.
264   */
265  public boolean hasNext() {
266    return !isBlank(getNextPageUrl());
267  }
268
269  /**
270   * provides the total count of elements, if FB provides them (API >= v2.0)
271   * 
272   * @return the total count of elements if present
273   * @since 1.6.16
274   */
275  public Long getTotalCount() {
276    return totalCount;
277  }
278  
279  public String getBeforeCursor() {
280      return beforeCursor;
281  }
282  
283  public String getAfterCursor() {
284      return afterCursor;
285  }
