023package com.restfb.scope;
024
025public enum ExtendedPermissions implements FacebookPermissions {
026
027  ADS_MANAGEMENT("ads_management"), //
028  ADS_READ("ads_read"), //
029
030  /**
031   * Provides access to the person's primary email address via the email property on the user object.
032   * 
033   * Do not spam users. Your use of email must comply with both Facebook policies and with the CAN-SPAM Act.
034   * 
035   * Note, even if you request the email permission it is not guaranteed you will get an email address. For example, if
036   * someone signed up for Facebook with a phone number instead of an email address, the email field may be empty.
037   * 
038   * <strong>Review</strong>
039   * 
040   * Your app may use this permission without review from Facebook.
041   */
042  EMAIL("email"), //
043
044  /**
045   * Enables your app to read a person's notifications and mark them as read.
046   * 
047   * This permission does not let you send notifications to a person.
048   * 
049   * <strong>Review</strong>
050   *
051   * If your app requests this permission Facebook will have to review how your app uses it.
052   */
053  MANAGE_NOTIFICATIONS("manage_notifications"), //
054
055  /**
056   * Enables your app to retrieve Page Access Tokens for the Pages and Apps that the person administrates.
057   * 
058   * <strong>Review</strong>
059   * 
060   * If your app requests this permission Facebook will have to review how your app uses it. You can grant this
061   * permission on behalf of people listed within the Roles section of your App's Dashboard without review by Facebook.
062   */
063  MANAGE_PAGES("manage_pages"), //
064
065  /**
066   * Provides access to publish Posts, Open Graph actions, achievements, scores and other activity on behalf of a person
067   * using your app.
068   * 
069   * Because this permission lets you publish on behalf of a user please read the Platform Policies to ensure you
070   * understand how to properly use this permission.
071   * 
072   * Your app does not need to request the publish_actions permission in order to use the Feed Dialog, the Requests
073   * Dialog or the Send Dialog
074   * 
075   * <strong>Review</strong>
076   * 
077   * If your app requests this permission Facebook will have to review how your app uses it.
078   */
079  PUBLISH_ACTIONS("publish_actions"), //
080  
081  /**
082   * Gives your app the ability to post, comment and like as any of the Pages managed by a person using your app. 
083   * 
084   * Publishing as an individual personal account is not possible with this permission. To post 
085   * as an individual, please see the <code>publish_actions</code> permission.
086   * 
087   * <strong>Review</strong>
088   * 
089   * If your app requests this permission Facebook will have to review how your app uses it.
090   * 
091   * since Graph API 2.3
092   */
093  PUBLISH_PAGES("publish_pages"), //
094
095  /**
096   * Provides access to the names of custom lists a person has created to organize their friends. This is useful for
097   * rendering an audience selector when someone is publishing stories to Facebook from your app.
098   * 
099   * This permission does not give access to a list of person's friends. If you want to access a person's friends who
100   * also use your app, you should use the user_friends permission.
101   * 
102   * This permission will also not help you invite a person's friends to use your app.
103   * 
104   * <strong>Review</strong>
105   * 
106   * If your app requests this permission Facebook will have to review how your app uses it.
107   * 
108   * @deprecated in Graph API 2.3 use READ_CUSTOM_FRIENDLISTS instead. If you use the Graph API before 2.3 ignore the deprecation warning
109   */
110  @Deprecated
111  READ_FRIENDLISTS("read_friendlists"), //
112  
113  /**
114   * Provides access to the names of custom lists a person has created to organize their friends. 
115   * 
116   * This is useful for rendering an audience selector when someone is publishing stories to Facebook from your app.
117   * 
118   * This permission does not give access to a list of person's friends. If you want to access a person's friends who
119   * also use your app, you should use the user_friends permission.
120   * 
121   * This permission will also not help you invite a person's friends to use your app.
122   * 
123   * This permission was called read_friendlists before v2.3.
124   * 
125   * <strong>Review</strong>
126   * 
127   * If your app requests this permission Facebook will have to review how your app uses it.
128   */
129  READ_CUSTOM_FRIENDLISTS("read_custom_friendlists"), //
130
131  /**
132   * Provides read-only access to the Insights data for Pages, Apps and web domains the person owns.
133   * 
134   * <strong>Review</strong>
135   * 
136   * If your app requests this permission Facebook will have to review how your app uses it.
137   */
138  READ_INSIGHTS("read_insights"), //
139
140  /**
141   * Provides the ability to read the messages in a person's Facebook Inbox through the inbox edge and the thread node.
142   * 
143   * <strong>Review</strong>
144   * 
145   * If your app requests this permission Facebook will have to review how your app uses it.
146   */
147  READ_MAILBOX("read_mailbox"), //
148
149  /**
150   * Provides the ability to read from the Page Inboxes of the Pages managed by a person. This permission is often used
151   * alongside the manage_pages permission.
152   * 
153   * This permission does not let your app read the page owner's mailbox. It only applies to the page's mailbox.
154   * 
155   * <strong>Review</strong>
156   * 
157   * If your app requests this permission Facebook will have to review how your app uses it.
158   */
159  READ_PAGE_MAILBOXES("read_page_mailboxes"), //
160
161  /**
162   * Provides access to read the posts in a person's News Feed, or the posts on their Profile.
163   * 
164   * <strong>Review</strong>
165   * 
166   * If your app requests this permission Facebook will have to review how your app uses it.
167   */
168  READ_STREAM("read_stream"), //
169
170  /**
171   * Provides the ability to set a person's attendee status on Facebook Events (eg attending, maybe, or declined).
172   * 
173   * This permission does not let you invite people to an event.
174   * 
175   * This permission does not let you update an event's details.
176   * 
177   * This permission does not let you create an event. There is no way to create an event via the API as of Graph API
178   * v2.0.
179   * 
180   * <strong>Review</strong>
181   * 
182   * If your app requests this permission Facebook will have to review how your app uses it.
183   */
184  RSVP_EVENT("rsvp_event");
185
186  ExtendedPermissions(String facebookPermissionString) {
187    this.permissionString = facebookPermissionString;
188  }
189
190  String permissionString;
191
192  @Override
193  public String getPermissionString() {
194    return this.permissionString;
195  }
196
197}
